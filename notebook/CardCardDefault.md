```python
sse_list = list()
silhouette_list = list()

max_k = 50
for k in range(2, max_k + 1):
    kmeans = KMeans(n_clusters=k, n_init=10, max_iter=50)
    kmeans.fit(X)
    
    sse = kmeans.inertia_
    sse_list.append(sse)
    silhouette = silhouette_score(X, kmeans.labels_)
    silhouette_list.append(silhouette)
```
